Bài 11: Đếm số chẵn

Viết chương trình đếm số lượng số chẵn trong danh sách 5 số người dùng nhập.

count = 0 for i in range(5): num = int(input(f"Nhập số thứ {i+1}: ")) if num % 2 == 0: count += 1 print("Số lượng số chẵn:", count)
# Cách 1: Sử dụng danh sách để lưu các số numbers = [] for i in range(5): num = int(input(f"Nhập số thứ {i+1}: ")) numbers.append(num) # Đếm số chẵn trong danh sách count = 0 for num in numbers: if num % 2 == 0: count += 1 print("Số lượng số chẵn:", count)
# Cách 2: Sử dụng vòng lặp while count = 0 i = 0 while i < 5: num = int(input(f"Nhập số thứ {i+1}: ")) if num % 2 == 0: count += 1 i += 1 print("Số lượng số chẵn:", count)
# Cách 3: Nhập tất cả số rồi đếm sau # Nhập 5 số numbers = [] for i in range(5): num = int(input(f"Nhập số thứ {i+1}: ")) numbers.append(num) # Sử dụng list comprehension để đếm số chẵn even_count = sum(1 for num in numbers if num % 2 == 0) print("Số lượng số chẵn:", even_count)
