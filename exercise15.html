Bài 15: Đếm số nguyên âm

Viết chương trình đếm số nguyên âm (a, e, i, o, u) trong chuỗi người dùng nhập.

s = input("Nhập chuỗi: ") vowels = "aeiou" count = 0 for char in s.lower(): if char in vowels: count += 1 print("Số nguyên âm:", count)
# Cách 1: Đếm từng nguyên âm riêng biệt s = input("Nhập chuỗi: ") s = s.lower() count_a = s.count('a') count_e = s.count('e') count_i = s.count('i') count_o = s.count('o') count_u = s.count('u') total = count_a + count_e + count_i + count_o + count_u print("Số nguyên âm:", total)
# Cách 2: Sử dụng list comprehension s = input("Nhập chuỗi: ") vowels = "aeiou" count = sum(1 for char in s.lower() if char in vowels) print("Số nguyên âm:", count)
# Cách 3: Sử dụng từ điển để đếm chi tiết s = input("Nhập chuỗi: ") vowels = {'a': 0, 'e': 0, 'i': 0, 'o': 0, 'u': 0} for char in s.lower(): if char in vowels: vowels[char] += 1 total = sum(vowels.values()) print("Số nguyên âm:", total) print("Chi tiết:") for vowel, count in vowels.items(): if count > 0: print(f"- Nguyên âm '{vowel}': {count}")
