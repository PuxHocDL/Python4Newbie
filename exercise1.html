Bài tập Python cơ bản

Bài 1: Tính tổng hai số

Viết chương trình yêu cầu nhập hai số nguyên và in tổng của chúng.

a = int(input("Nhập số thứ nhất: ")) b = int(input("Nhập số thứ hai: ")) tong = a + b print("Tổng là:", tong)
# Cách 1: Sử dụng hàm def tinh_tong(a, b): return a + b so_thu_nhat = int(input("Nhập số thứ nhất: ")) so_thu_hai = int(input("Nhập số thứ hai: ")) tong = tinh_tong(so_thu_nhat, so_thu_hai) print("Tổng là:", tong)
# Cách 2: Giải pháp ngắn gọn print("Tổng là:", int(input("Nhập số thứ nhất: ")) + int(input("Nhập số thứ hai: ")))
# Cách 3: Sử dụng danh sách numbers = [] numbers.append(int(input("Nhập số thứ nhất: "))) numbers.append(int(input("Nhập số thứ hai: "))) print("Tổng là:", sum(numbers))
