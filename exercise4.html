Bài 4: Tính giai thừa

Viết hàm đệ quy tính giai thừa của một số.

def factorial(n): if n == 0: return 1 return n * factorial(n-1) n = int(input("Nhập một số: ")) print(f"Giai thừa của {n} là:", factorial(n))
# Cách 1: Sử dụng vòng lặp (không đệ quy) def factorial(n): result = 1 for i in range(1, n + 1): result *= i return result n = int(input("Nhập một số: ")) print(f"Giai thừa của {n} là:", factorial(n))
# Cách 2: Sử dụng hàm có sẵn từ thư viện math import math n = int(input("Nhập một số: ")) print(f"Giai thừa của {n} là:", math.factorial(n))
# Cách 3: Sử dụng functools.reduce và lambda from functools import reduce def factorial(n): if n == 0: return 1 return reduce(lambda x, y: x * y, range(1, n + 1)) n = int(input("Nhập một số: ")) print(f"Giai thừa của {n} là:", factorial(n))
