Bài 7: Tính tổng dãy số

Tính tổng các số từ 1 đến n bằng đệ quy.

def sum_series(n): if n == 1: return 1 return n + sum_series(n-1) n = int(input("Nhập n: ")) print(f"Tổng từ 1 đến {n} là:", sum_series(n))
# Cách 1: Sử dụng vòng lặp for n = int(input("Nhập n: ")) total = 0 for i in range(1, n + 1): total += i print(f"Tổng từ 1 đến {n} là:", total)
# Cách 2: Sử dụng công thức toán học n = int(input("Nhập n: ")) total = n * (n + 1) // 2 print(f"Tổng từ 1 đến {n} là:", total)
# Cách 3: Sử dụng hàm sum() và range() n = int(input("Nhập n: ")) total = sum(range(1, n + 1)) print(f"Tổng từ 1 đến {n} là:", total)
