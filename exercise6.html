Bài 6: Đếm ký tự

Đếm số ký tự không tính khoảng trắng trong chuỗi.

def count_chars(s): return len(s.replace(" ", "")) s = input("Nhập chuỗi: ") print("Số ký tự:", count_chars(s))
# Cách 1: Sử dụng vòng lặp for s = input("Nhập chuỗi: ") count = 0 for char in s: if char != " ": count += 1 print("Số ký tự:", count)
# Cách 2: Sử dụng list comprehension s = input("Nhập chuỗi: ") count = sum(1 for char in s if char != " ") print("Số ký tự:", count)
# Cách 3: Sử dụng phương thức split() và join() s = input("Nhập chuỗi: ") no_spaces = "".join(s.split()) print("Số ký tự:", len(no_spaces))
