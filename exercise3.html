Bài 3: Ghi danh sách số

Nhập 3 số và in danh sách (mô phỏng ghi file).

numbers = [] for i in range(3): num = int(input(f"Nhập số {i+1}: ")) numbers.append(num) print("Danh sách đã nhập:", numbers)
# Cách 1: Sử dụng list comprehension numbers = [int(input(f"Nhập số {i+1}: ")) for i in range(3)] print("Danh sách đã nhập:", numbers)
# Cách 2: Sử dụng vòng lặp while numbers = [] i = 0 while i < 3: num = int(input(f"Nhập số {i+1}: ")) numbers.append(num) i += 1 print("Danh sách đã nhập:", numbers)
# Cách 3: Nhập cùng lúc nhiều số và chuyển thành danh sách print("Nhập 3 số, cách nhau bởi dấu cách:") numbers = list(map(int, input().split()))[:3] # Giới hạn 3 số print("Danh sách đã nhập:", numbers)
