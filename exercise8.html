Bài 8: Đảo ngược chuỗi

Viết chương trình đảo ngược chuỗi người dùng nhập.

s = input("Nhập chuỗi: ") print("Chuỗi đảo ngược:", s[::-1])
# Cách 1: Sử dụng vòng lặp for s = input("Nhập chuỗi: ") reversed_s = "" for char in s: reversed_s = char + reversed_s print("Chuỗi đảo ngược:", reversed_s)
# Cách 2: Sử dụng hàm reversed() và join() s = input("Nhập chuỗi: ") reversed_s = "".join(reversed(s)) print("Chuỗi đảo ngược:", reversed_s)
# Cách 3: Chuyển thành list, đảo ngược và join s = input("Nhập chuỗi: ") char_list = list(s) char_list.reverse() reversed_s = "".join(char_list) print("Chuỗi đảo ngược:", reversed_s)
