Bài 2: Kiểm tra số chẵn/lẻ

Viết hàm kiểm tra số chẵn/lẻ và in kết quả.

def is_even(n): return n % 2 == 0 n = int(input("Nhập một số: ")) if is_even(n): print("Chẵn") else: print("Lẻ")
# Cách 1: Không sử dụng hàm n = int(input("Nhập một số: ")) if n % 2 == 0: print("Chẵn") else: print("Lẻ")
# Cách 2: Sử dụng toán tử bit (bitwise AND) def is_even(n): return (n & 1) == 0 # Kiểm tra bit cuối cùng n = int(input("Nhập một số: ")) print("Chẵn" if is_even(n) else "Lẻ")
# Cách 3: Sử dụng hàm lambda và toán tử ba ngôi is_even = lambda x: x % 2 == 0 n = int(input("Nhập một số: ")) print("Chẵn" if is_even(n) else "Lẻ")
